46template<typename _Scalar, int _Rows, int _Cols>
47void to_json(json& j, const Matrix<_Scalar, _Rows, _Cols>& matrix)
48{
49 for (int r = 0; r < matrix.rows(); r++)
50 {
51 for (int c = 0; c < matrix.cols(); c++)
52 {
53 j[std::to_string(r)][std::to_string(c)] = matrix(r, c);
54 }
55 }
56}

64template<typename _Scalar, int _Rows, int _Cols>
65void from_json(const json& j, Matrix<_Scalar, _Rows, _Cols>& matrix)
66{
67 if constexpr (_Rows == -1 || _Cols == -1)
68 {
69 int rows = -1;
70 int cols = -1;
71 for (int r = 0; j.contains(std::to_string(r)); r++)
72 {
73 rows = std::max(r, rows);
74 for (int c = 0; j.at(std::to_string(r)).contains(std::to_string(c)); c++)
75 {
76 cols = std::max(c, cols);
77 }
78 }
79 matrix = Eigen::Matrix<_Scalar, _Rows, _Cols>::Zero(rows + 1, cols + 1);
80 }
81
82 for (int r = 0; j.contains(std::to_string(r)); r++) // NOLINT(readability-misleading-indentation)
83 {
84 for (int c = 0; j.at(std::to_string(r)).contains(std::to_string(c)); c++)
85 {
86 j.at(std::to_string(r)).at(std::to_string(c)).get_to(matrix(r, c));
87 }
88 }
89}

100template<typename Derived>
101void removeRows(Eigen::DenseBase<Derived>& matrix, size_t index, size_t length)
102{
103 INS_ASSERT_USER_ERROR(static_cast<size_t>(matrix.rows()) >= index + length, "Tried to remove rows which do not exist");
104
105 std::vector<int> indicesToKeep;
106 indicesToKeep.reserve(static_cast<size_t>(matrix.rows()) - length);
107 for (int i = 0; i < static_cast<int>(index); i++) { indicesToKeep.push_back(i); }
108 for (int i = static_cast<int>(index + length); i < matrix.rows(); i++) { indicesToKeep.push_back(i); }
109
110 matrix = matrix(indicesToKeep, Eigen::all).eval();
111}

116template<typename Derived>
117void removeRows(Eigen::DenseBase<Derived>& matrix, const std::vector<int>& rowIndices)
118{
119 std::vector<int> rowIndicesToKeep;
120 rowIndicesToKeep.reserve(static_cast<size_t>(matrix.rows()) - rowIndices.size());
121 for (int i = 0; i < matrix.rows(); i++)
122 {
123 if (std::find(rowIndices.begin(), rowIndices.end(), i) == rowIndices.end())
124 {
125 rowIndicesToKeep.push_back(i);
126 }
127 }
128
129 matrix = matrix(rowIndicesToKeep, Eigen::all).eval();

178 INS_ASSERT_USER_ERROR(static_cast<size_t>(matrix.cols()) >= col + cols, "Tried to remove cols which do not exist");

182 for (int i = 0; i < static_cast<int>(row); i++) { rowsToKeep.push_back(i); }
183 for (int i = static_cast<int>(row + rows); i < matrix.rows(); i++) { rowsToKeep.push_back(i); }
