127 void setCovarianceMatrix(const KeyedMatrixXd<States::StateKeyTypes, States::StateKeyTypes>& Q)
128 {
129 _e_covarianceMatrix = Q;
130 std::vector<States::StateKeyTypes> interSysErrs;
131 std::vector<States::StateKeyTypes> interSysDrifts;
132 for (const auto& key : Q.colKeys())
133 {
134 if (std::get_if<States::InterSysErr>(&key))
135 {
136 interSysErrs.emplace_back(key);
137 }
138 if (std::get_if<States::InterSysDrift>(&key))
139 {
140 interSysDrifts.emplace_back(key);
141 }
142 }
143
144 n_CovarianceMatrix_e(interSysErrs, interSysDrifts);
145 }
146
150 void setCovarianceMatrix(const std::vector<States::StateKeyTypes>& interSysErrs, const std::vector<States::StateKeyTypes>& interSysDrifts)
151 {
152 if (!_e_covarianceMatrix.hasAnyCols(States::PosVelRecvClk)) // creates states for first use
153 {
154 _e_covarianceMatrix.addRowsCols(States::PosVelRecvClk, States::PosVelRecvClk);
155 _e_covarianceMatrix.addRowsCols(interSysErrs, interSysErrs);
156 _e_covarianceMatrix.addRowsCols(interSysDrifts, interSysDrifts);
157 }
158 else // check whether interSys keys and states in _e_covariance are the same and remove or add accordingly (e. g. due to elevation mask)
159 {
160 for (const auto& key : _e_covarianceMatrix.colKeys())
161 {
162 if (std::get_if<States::InterSysErr>(&key))
163 {
164 if (std::find(interSysErrs.begin(), interSysErrs.end(), key) != interSysErrs.end())
165 {
166 continue;
167 }
168 _e_covarianceMatrix.removeCol(key);
169 _e_covarianceMatrix.removeRow(key);
170 }
171 }
172 for (const auto& key : _e_covarianceMatrix.colKeys())
173 {
174 if (std::get_if<States::InterSysDrift>(&key))
175 {
176 if (std::find(interSysDrifts.begin(), interSysDrifts.end(), key) != interSysDrifts.end())
177 {
178 continue;
179 }
180 _e_covarianceMatrix.removeCol(key);
181 _e_covarianceMatrix.removeRow(key);
182 }
183 }
184 for (size_t i = 0; i < interSysErrs.size(); i++)
185 {
186 if (std::find(_e_covarianceMatrix.colKeys().begin(), _e_covarianceMatrix.colKeys().end(), interSysErrs.at(i)) == _e_covarianceMatrix.colKeys().end())
187 {
188 _e_covarianceMatrix.addCol(interSysErrs.at(i));
189 _e_covarianceMatrix.addRow(interSysErrs.at(i));
190 _e_covarianceMatrix.addCol(interSysDrifts.at(i));
191 _e_covarianceMatrix.addRow(interSysDrifts.at(i));
192 }
193 }
194 }
195
196 _e_covarianceMatrix(States::PosRecvClkErr, States::PosRecvClkErr) = _e_positionClockErrorCovarianceMatrix(States::PosRecvClkErr, States::PosRecvClkErr);
197 _e_covarianceMatrix(interSysErrs, interSysErrs) = _e_positionClockErrorCovarianceMatrix(interSysErrs, interSysErrs);
198 _e_covarianceMatrix(States::PosRecvClkErr, interSysErrs) = _e_positionClockErrorCovarianceMatrix(States::PosRecvClkErr, interSysErrs);
199 _e_covarianceMatrix(interSysErrs, States::PosRecvClkErr) = _e_positionClockErrorCovarianceMatrix(interSysErrs, States::PosRecvClkErr);
200
201 if (_e_velocityClockDriftCovarianceMatrix.cols() > 0)
202 {
203 _e_covarianceMatrix(States::VelRecvClkDrift, States::VelRecvClkDrift) = _e_velocityClockDriftCovarianceMatrix(States::VelRecvClkDrift, States::VelRecvClkDrift);
204 _e_covarianceMatrix(interSysDrifts, interSysDrifts) = _e_velocityClockDriftCovarianceMatrix(interSysDrifts, interSysDrifts);
205 _e_covarianceMatrix(States::VelRecvClkDrift, interSysDrifts) = _e_velocityClockDriftCovarianceMatrix(States::VelRecvClkDrift, interSysDrifts);
206 _e_covarianceMatrix(interSysDrifts, States::VelRecvClkDrift) = _e_velocityClockDriftCovarianceMatrix(interSysDrifts, States::VelRecvClkDrift);
207 }
208
209 n_CovarianceMatrix_e(interSysErrs, interSysDrifts);
210 }

215 void n_CovarianceMatrix_e(const std::vector<States::StateKeyTypes>& interSysErrs, const std::vector<States::StateKeyTypes>& interSysDrifts)
216 {
217 _n_covarianceMatrix = _e_covarianceMatrix;
218
219 _n_covarianceMatrix(States::PosVel, States::PosVel).setZero();
220 Eigen::Vector3d lla_pos = lla_position();
221 Eigen::Quaterniond n_Quat_e = trafo::n_Quat_e(lla_pos(0), lla_pos(1));
222 _n_covarianceMatrix(States::Pos, States::Pos) = n_Quat_e.toRotationMatrix() * _e_covarianceMatrix(States::Pos, States::Pos) * n_Quat_e.conjugate().toRotationMatrix(); // variance of position
223 _n_covarianceMatrix(States::Vel, States::Vel) = n_Quat_e.toRotationMatrix() * _e_covarianceMatrix(States::Vel, States::Vel) * n_Quat_e.toRotationMatrix(); // variance of velocity
224
225 _n_covarianceMatrix(States::PosVel, States::RecvClk).setZero();
226 _n_covarianceMatrix(States::RecvClk, States::PosVel).setZero();
227 _n_covarianceMatrix(States::PosVel, interSysErrs).setZero();
228 _n_covarianceMatrix(interSysErrs, States::PosVel).setZero();
229 _n_covarianceMatrix(States::PosVel, interSysDrifts).setZero();
230 _n_covarianceMatrix(interSysDrifts, States::PosVel).setZero();
231 }
