35Eigen::Matrix<typename Derived::Scalar, 3, 3> skewSymmetricMatrix(const Eigen::MatrixBase<Derived>& a)
36{
37 INS_ASSERT_USER_ERROR(a.cols() == 1, "Given Eigen Object must be a vector");
38 INS_ASSERT_USER_ERROR(a.rows() == 3, "Given Vector must have 3 Rows");

52template<typename Derived>
53Eigen::Matrix<typename Derived::Scalar, 3, 3> skewSymmetricMatrixSquared(const Eigen::MatrixBase<Derived>& a)
54{
55 INS_ASSERT_USER_ERROR(a.cols() == 1, "Given Eigen Object must be a vector");
56 INS_ASSERT_USER_ERROR(a.rows() == 3, "Given Vector must have 3 Rows");
57
58 Eigen::Matrix<typename Derived::Scalar, 3, 3> skewMat2;
59 skewMat2 << std::pow(a(2), 2) + std::pow(a(1), 2), a(0) * a(1), a(0) * a(2),
60 a(0) * a(1), std::pow(a(2), 2) + std::pow(a(0), 2), a(1) * a(2),
61 a(0) * a(2), a(1) * a(2), std::pow(a(0), 2) + std::pow(a(1), 2);
62
63 return skewMat2;
