126template<typename Scalar, typename RowKeyType, int Cols>
127class KeyedMatrixRows<Scalar, RowKeyType, Eigen::Dynamic, Cols> : public KeyedMatrixRowsBase<Scalar, RowKeyType, Eigen::Dynamic, Cols>
128{
129 public:
132 void addRow(const RowKeyType& rowKey) { addRows({ rowKey }); }
133
136 void addRows(const std::vector<RowKeyType>& rowKeys)
137 {
138 INS_ASSERT_USER_ERROR(!this->hasAnyRows(rowKeys), "You cannot add a row key which is already in the matrix.");
139 INS_ASSERT_USER_ERROR(std::unordered_set<RowKeyType>(rowKeys.begin(), rowKeys.end()).size() == rowKeys.size(), "Each row key must be unique");
140
141 auto initialSize = static_cast<Eigen::Index>(this->rowIndices.size());
142 for (const auto& rowKey : rowKeys) { this->rowIndices.insert({ rowKey, static_cast<Eigen::Index>(this->rowIndices.size()) }); }
143 this->rowKeysVector.reserve(this->rowKeysVector.size() + rowKeys.size());
144 std::copy(rowKeys.begin(), rowKeys.end(), std::back_inserter(this->rowKeysVector));
145 auto finalSize = static_cast<Eigen::Index>(this->rowIndices.size());
146
147 if (finalSize > initialSize)
148 {
149 this->matrix.conservativeResize(finalSize, Eigen::NoChange);
150 this->matrix.block(initialSize, 0, finalSize - initialSize, this->matrix.cols()) = Eigen::MatrixX<Scalar>::Zero(finalSize - initialSize, this->matrix.cols());
151 }
152 this->rowSlice.reserve(this->rowKeysVector.size());
153 }
154
157 void removeRow(const RowKeyType& rowKey) { removeRows({ rowKey }); }
158
161 void removeRows(const std::vector<RowKeyType>& rowKeys)
162 {
163 std::vector<int> indices;
164 for (const auto& rowKey : rowKeys)
165 {
166 auto iter = std::find_if(this->rowIndices.begin(), this->rowIndices.end(), [&](const auto& item) { return item.first == rowKey; });
167 INS_ASSERT_USER_ERROR(iter != this->rowIndices.end(), "You tried removing a row key, which did not exist.");
168 if (iter != this->rowIndices.end())
169 {
170 indices.push_back(static_cast<int>(iter->second));
171 }
172 }
173 NAV::removeRows(this->matrix, indices);
174
175 for (const auto& rowKey : rowKeys)
176 {
177 auto iter = std::find_if(this->rowIndices.begin(), this->rowIndices.end(), [&](const auto& item) { return item.first == rowKey; });
178 if (iter != this->rowIndices.end())
179 {
180 std::erase_if(this->rowKeysVector, [&](const auto& item) { return item == rowKey; });
181
182 auto idx = iter->second;
183 for (auto& rowIndex : this->rowIndices)
184 {
185 if (rowIndex.second > idx) { rowIndex.second--; }
186 }
187 this->rowIndices.erase(iter);
188 }
189 }
190 }
191};

2144 {

2146 INS_ASSERT_USER_ERROR(!this->hasAnyCols(colKeys), "You cannot add a col key which is already in the matrix.");
2147 INS_ASSERT_USER_ERROR(std::unordered_set<RowKeyType>(rowKeys.begin(), rowKeys.end()).size() == rowKeys.size(), "Each row key must be unique");
2148 INS_ASSERT_USER_ERROR(std::unordered_set<ColKeyType>(colKeys.begin(), colKeys.end()).size() == colKeys.size(), "Each col key must be unique");
2149
2150 auto initialRowSize = static_cast<Eigen::Index>(this->rowIndices.size());
2151 for (const auto& rowKey : rowKeys) { this->rowIndices.insert({ rowKey, static_cast<Eigen::Index>(this->rowIndices.size()) }); }
2152 this->rowKeysVector.reserve(this->rowKeysVector.size() + rowKeys.size());

2154 auto finalRowSize = static_cast<Eigen::Index>(this->rowIndices.size());
2155
2156 auto initialColSize = static_cast<Eigen::Index>(this->colIndices.size());
2157 for (const auto& colKey : colKeys) { this->colIndices.insert({ colKey, static_cast<Eigen::Index>(this->colIndices.size()) }); }
2158 this->colKeysVector.reserve(this->colKeysVector.size() + colKeys.size());

2160 auto finalColSize = static_cast<Eigen::Index>(this->colIndices.size());
